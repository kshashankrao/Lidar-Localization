cmake_minimum_required(VERSION 3.14)
project(lidar_localization LANGUAGES C CXX)

# 1. Enforce C++17 (Required for std::filesystem in your data loader)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF) # Disables compiler-specific extensions for portability

# Eigen 
find_package(Eigen3 REQUIRED)

# PCL
find_package(MPI REQUIRED)
find_package(PCL REQUIRED COMPONENTS common io filters registration visualization)
find_package(nlohmann_json REQUIRED)
find_package(OpenMP REQUIRED)
add_definitions(${PCL_DEFINITIONS})
include_directories(${PCL_INCLUDE_DIRS})
link_directories(${PCL_LIBRARY_DIRS})

# Core library sources (excluding main.cpp)
file(GLOB_RECURSE LIB_SOURCES "src/config/*.cpp" "src/dataloader/*.cpp" "src/ekf/*.cpp" "src/icp/*.cpp" "src/localization/*.cpp" "src/loam/*.cpp")

add_library(lidar_localization_lib STATIC ${LIB_SOURCES})
target_include_directories(lidar_localization_lib PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/src)
target_link_libraries(lidar_localization_lib PUBLIC 
    Eigen3::Eigen
    ${PCL_LIBRARIES}
    MPI::MPI_C
    nlohmann_json::nlohmann_json
    OpenMP::OpenMP_CXX
)

# Set output directory for executable to build/bin BEFORE creating the executable
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# Main executable
add_executable(lidar_localization src/main.cpp)
target_link_libraries(lidar_localization PRIVATE lidar_localization_lib)

# Copy config directory to build/bin after build
add_custom_command(TARGET lidar_localization POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/config
    ${CMAKE_BINARY_DIR}/bin/config
    COMMENT "Copying config folder to build/bin"
)

# --- Unit Testing (Google Test) ---
include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)
# For Windows: Prevent overriding the parent project's compiler/linker settings
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

enable_testing()

file(GLOB_RECURSE TEST_SOURCES "tests/*.cpp")
if(TEST_SOURCES)
    add_executable(unit_tests ${TEST_SOURCES})
    target_link_libraries(unit_tests PRIVATE 
        lidar_localization_lib 
        gtest_main
    )
    include(GoogleTest)
    gtest_discover_tests(unit_tests)
endif()
